.filename -}} {{- end -}} {{- $params := dict -}} {{- $query := split $url.RawQuery "&" -}} {{- range $item := $query -}} {{- $pair := split $item "=" -}} {{- $params = merge $params (dict (index $pair 0) (index $pair 1)) -}} {{- end -}} {{- if index $params "width" -}} {{- $width = $params.width -}} {{- end -}} {{- if index $params "height" -}} {{- $height = $params.height -}} {{- end -}} {{- safeURL -}} {{- else if and (ne $height false) $pageResource }} {{- $resize := (printf "x%s" (strings.TrimSuffix "px" $height)) -}} {{- $img = $img.Resize $resize -}} {{- $link = $img.Permalink |
